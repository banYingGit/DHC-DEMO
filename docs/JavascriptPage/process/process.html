在目标元素内动态创建一进度条，静态显示

  • Demo
  • html
  • javaScript

  • <div id="proA"></div>

  • <script>

    $(function () {

    var proA = $('

    ').process({

    element: '#proA',

    max:'88'

    }).on('process',function(){

    console.log(arguments)

    });

    proA.trigger('context', '正在加载');

    proA.process('value', 50)

    })

    </script>

2.自动加载

当前元素自动转换为进度条结构

  • Demo
  • html
  • javaScript

  • <div id="proB"></div>

  • <script>

    $(function () {

    var myPro,

    i = 0,

    proB = $('#proB').process({

    max: 90

    }).on('end', function () {

    clearInterval(myPro)

    });

    myPro = setInterval(function () {

    proB.process('value', i++).trigger('context', '正在加载' + i)

    }, 100)

    })

    </script>
